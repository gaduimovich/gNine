  (define split-2 (+ split-1 W gutter))
  (canvas (+ (* 3 W) (* 2 gutter)) H 3
    (if (< j W)
      (if (== c 0) (A 0 0) 0)

        (if (< j (+ split-1 W))

; Suggested preview command:

  (define start-pipe2-x (+ start-pipe1-x pipe-spacing))

    (lambda (idx)

      (define next-tick-base (+ tick 1))

      (define accelerated-vy
        (if (> alive 0)

      (define hit-pipe2

; Runtime Pong demo.
; This version uses iterate-until so the game state evolves indefinitely:
;   frame, ball-x, ball-y, vx, vy, left-paddle-y, right-paddle-y,
;   left-score, right-score
;
; Preview controls:
;   left paddle: Up / Down
; The right paddle remains AI-controlled.
;
; Render with:
;   ./gnine --runtime --emit-frames=runtime_pong.png \
;     runtime_pong.psm \
;     runtime_pong_final.png

(iterate-until 100
  (()
    (define W 240)
    (define H 135)
    (tuple
      (canvas W H 3
        (if (== c 0) 0.04
          (if (== c 1) 0.06 0.12)))
      (/ W 2)
      (/ H 2)
      (max 3 (int (/ W 90)))
      (max 2 (int (/ H 120)))
      (/ H 2)
      (/ H 2)
      0
      0))

  (((frame ball-x ball-y vx vy left-y right-y left-score right-score))
    (define W (width frame))
    (define H (height frame))
    (define mid-x (/ W 2))
    (define paddle-half-h (max 5 (int (/ H 9))))
    (define paddle-half-w (max 2 (int (/ W 90))))
    (define paddle-left-x (int (/ W 14)))
    (define paddle-right-x (- W paddle-left-x))
    (define paddle-speed (max 2 (int (/ H 80))))
    (define ball-half (max 2 (int (/ W 110))))
    (define serve-vx (max 3 (int (/ W 90))))
    (define serve-vy (max 2 (int (/ H 120))))
    (define wall-thickness (max 1 (int (/ H 150))))
    (define paddle-min (+ wall-thickness paddle-half-h))
    (define paddle-max (- H wall-thickness paddle-half-h 1))
    (define top-limit (+ wall-thickness ball-half))
    (define bottom-limit (- H wall-thickness ball-half 1))
    (define net-half-w (max 1 (int (/ W 220))))
    (define net-half-h (max 3 (int (/ H 18))))
    (define score-bar-half-h (max 3 (int (/ H 40))))
    (define score-bar-max-half-w (max 16 (int (/ W 10))))
    (define score-y (+ wall-thickness score-bar-half-h 8))
    (define left-score-center (- mid-x (max 18 (int (/ W 7)))))
    (define right-score-center (+ mid-x (max 18 (int (/ W 7)))))

    (define left-input (- key-down key-up))

    (define moved-left-y
      (clamp
        (+ left-y
           (* paddle-speed left-input))
        paddle-min
        paddle-max))

    (define moved-right-y
      (clamp
        (+ right-y (clamp (- ball-y right-y) (- paddle-speed) paddle-speed))
        paddle-min
        paddle-max))

    (define trial-x (+ ball-x vx))
    (define trial-y (+ ball-y vy))
    (define left-scored (< trial-x 0))
    (define right-scored (>= trial-x W))

    (define clamped-y
      (clamp trial-y top-limit bottom-limit))

    (define left-hit
      (and (< vx 0)
           (<= (abs (- clamped-y moved-left-y)) paddle-half-h)
           (<= trial-x (+ paddle-left-x paddle-half-w ball-half))))

    (define right-hit
      (and (> vx 0)
           (<= (abs (- clamped-y moved-right-y)) paddle-half-h)
